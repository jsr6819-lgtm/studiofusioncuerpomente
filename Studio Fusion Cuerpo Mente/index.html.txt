    <script>
        // Datos
        let users = JSON.parse(localStorage.getItem('pilatesUsers')) || [
            { id: 'admin', name: 'Administrador', email: 'admin@fusion.com', password: 'admin123', isAdmin: true }
        ];
        let appointments = JSON.parse(localStorage.getItem('pilatesAppointments')) || [];
        let currentUser = JSON.parse(localStorage.getItem('pilatesCurrentUser')) || null;

        // Elementos
        const authNav = document.getElementById('auth-nav');
        const userNav = document.getElementById('user-nav');
        const usernameDisplay = document.getElementById('username-display');
        const appointmentsList = document.getElementById('appointments-list');
        const adminAppointmentsList = document.getElementById('admin-appointments-list');
        const loginForm = document.getElementById('login-form');
        const registerForm = document.getElementById('register-form');
        const appointmentForm = document.getElementById('appointment-form');
        const logoutBtn = document.getElementById('logout-btn');
        const confirmCancel = document.getElementById('confirm-cancel');

        let appointmentToCancel = null;

        // Inicializar
        function init() {
            updateUI();
            renderAppointments();
            setupEventListeners();
        }

        // Actualizar UI
        function updateUI() {
            if (currentUser) {
                authNav.style.display = 'flex';
                userNav.style.display = 'flex';
                usernameDisplay.textContent = currentUser.name;
                if (currentUser.isAdmin) {
                    document.getElementById('admin-panel').style.display = 'block';
                    renderAdminAppointments();
                } else {
                    document.getElementById('admin-panel').style.display = 'none';
                }
            } else {
                authNav.style.display = 'flex';
                userNav.style.display = 'none';
                document.getElementById('admin-panel').style.display = 'none';
            }
        }

        // Listeners
        function setupEventListeners() {
            // Registro
            registerForm.addEventListener('submit', e => {
                e.preventDefault();
                const name = document.getElementById('register-name').value;
                const email = document.getElementById('register-email').value;
                const password = document.getElementById('register-password').value;
                const confirm = document.getElementById('register-confirm-password').value;

                if (password !== confirm) return alert('Las contraseñas no coinciden');
                if (users.some(u => u.email === email)) return alert('Este correo ya está registrado');

                const user = { id: Date.now(), name, email, password, isAdmin: false };
                users.push(user);
                localStorage.setItem('pilatesUsers', JSON.stringify(users));
                alert('✅ Registro exitoso');
                registerForm.reset();
                bootstrap.Modal.getInstance(document.getElementById('registerModal')).hide();
            });

            // Login
            loginForm.addEventListener('submit', e => {
                e.preventDefault();
                const email = document.getElementById('login-email').value;
                const password = document.getElementById('login-password').value;
                const isAdmin = document.getElementById('admin-login').checked;

                const user = users.find(u => u.email === email && u.password === password && u.isAdmin === isAdmin);
                if (user) {
                    currentUser = user;
                    localStorage.setItem('pilatesCurrentUser', JSON.stringify(currentUser));
                    updateUI();
                    renderAppointments();
                    loginForm.reset();
                    bootstrap.Modal.getInstance(document.getElementById('loginModal')).hide();
                } else {
                    alert('Credenciales inválidas');
                }
            });

            // Logout
            logoutBtn.addEventListener('click', () => {
                currentUser = null;
                localStorage.removeItem('pilatesCurrentUser');
                updateUI();
                renderAppointments();
            });

            // Reservar
            appointmentForm.addEventListener('submit', e => {
                e.preventDefault();
                if (!currentUser || currentUser.isAdmin) return alert('Inicia sesión como usuario');

                const date = document.getElementById('appointment-date').value;
                const time = document.getElementById('appointment-time').value;
                const classType = document.getElementById('class-type').value;

                if (appointments.some(a => a.userId === currentUser.id && a.date === date && a.time === time)) {
                    return alert('Ya tienes una cita a esa hora');
                }

                const appt = {
                    id: Date.now(),
                    userId: currentUser.id,
                    userName: currentUser.name,
                    date,
                    time,
                    classType
                };
                appointments.push(appt);
                localStorage.setItem('pilatesAppointments', JSON.stringify(appointments));
                alert('✅ Cita reservada con éxito');
                appointmentForm.reset();
                renderAppointments();
            });
        }

        // Renderizar citas del usuario
        function renderAppointments() {
            if (!currentUser) {
                appointmentsList.innerHTML = '<p class="text-center text-muted">Inicia sesión para ver tus citas.</p>';
                return;
            }
            if (currentUser.isAdmin) {
                appointmentsList.innerHTML = '<p class="text-center text-muted">Panel de administrador activo.</p>';
                return;
            }

            const userApps = appointments
                .filter(a => a.userId === currentUser.id)
                .sort((a, b) => new Date(a.date + ' ' + a.time) - new Date(b.date + ' ' + b.time));

            if (userApps.length === 0) {
                appointmentsList.innerHTML = '<p class="text-center text-muted">No tienes citas programadas.</p>';
                return;
            }

            appointmentsList.innerHTML = '';
            userApps.forEach(a => {
                const date = new Date(a.date).toLocaleDateString('es-ES', { weekday: 'short', day: '2-digit', month: 'short' });
                const card = document.createElement('div');
                card.className = 'appointment-card card mb-3';
                card.innerHTML = `
                    <div class="card-body">
                        <div class="d-flex justify-content-between align-items-center">
                            <div>
                                <h6 class="card-title mb-1">${a.classType}</h6>
                                <p class="card-text mb-0">
                                    <small>
                                        <i class="far fa-calendar"></i> ${date} | 
                                        <i class="far fa-clock"></i> ${a.time}
                                    </small>
                                </p>
                            </div>
                            <button class="btn btn-sm btn-outline-danger" onclick="openCancelModal(${a.id}, '${a.classType}', '${date}', '${a.time}')">
                                <i class="fas fa-trash"></i>
                            </button>
                        </div>
                    </div>
                `;
                appointmentsList.appendChild(card);
            });
        }

        // Renderizar citas del admin
        function renderAdminAppointments() {
            if (!currentUser?.isAdmin) return;
            adminAppointmentsList.innerHTML = '';

            if (appointments.length === 0) {
                adminAppointmentsList.innerHTML = '<tr><td colspan="5" class="text-center text-muted">No hay citas</td></tr>';
                return;
            }

            [...appointments].sort((a, b) => new Date(b.date) - new Date(a.date)).forEach(a => {
                const tr = document.createElement('tr');
                tr.innerHTML = `
                    <td>${a.userName}</td>
                    <td>${a.date}</td>
                    <td>${a.time}</td>
                    <td>${a.classType}</td>
                    <td>
                        <button class="btn btn-sm btn-danger" onclick="deleteAppointment(${a.id})">
                            <i class="fas fa-trash"></i>
                        </button>
                    </td>
                `;
                adminAppointmentsList.appendChild(tr);
            });
        }

        // Abrir modal de anulación
        window.openCancelModal = function(id, classType, date, time) {
            document.getElementById('cancel-info').textContent = `${classType} - ${date} a las ${time}`;
            appointmentToCancel = id;
            new bootstrap.Modal(document.getElementById('cancelModal')).show();
        };

        // Confirmar anulación
        confirmCancel.addEventListener('click', () => {
            if (appointmentToCancel) {
                appointments = appointments.filter(a => a.id !== appointmentToCancel);
                localStorage.setItem('pilatesAppointments', JSON.stringify(appointments));
                renderAppointments();
                renderAdminAppointments();
                new bootstrap.Modal(document.getElementById('cancelModal')).hide();
            }
        });

        // Eliminar cita (admin)
        window.deleteAppointment = function(id) {
            if (confirm('¿Eliminar esta cita?')) {
                appointments = appointments.filter(a => a.id !== id);
                localStorage.setItem('pilatesAppointments', JSON.stringify(appointments));
                renderAppointments();
                renderAdminAppointments();
            }
        };

        // Iniciar
        document.addEventListener('DOMContentLoaded', init);
    </script>
